// Úprava funkce pro tvorbu ovoce v Three.js function createStrawberry() { // Tělo jahody - deformovaný kužel const strawberryGeo = new THREE.SphereGeometry(0.2, 16, 16); strawberryGeo.scale(0.8, 1.2, 0.8); // Protáhneme do tvaru jahody const strawberryMat = new THREE.MeshPhongMaterial({ color: 0xff0000, shininess: 50, bumpScale: 0.02 }); const strawberry = new THREE.Mesh(strawberryGeo, strawberryMat); // Přidáme lístky (zelený disk navrchu) const leafGeo = new THREE.CircleGeometry(0.1, 8); const leafMat = new THREE.MeshBasicMaterial({ color: 0x228b22, side: THREE.DoubleSide }); const leaves = new THREE.Mesh(leafGeo, leafMat); leaves.rotation.x = Math.PI / 2; leaves.position.y = 0.2; strawberry.add(leaves); return strawberry; } function createBlueberry() { // Borůvka - kulička s tmavším vrškem const blueberryGeo = new THREE.SphereGeometry(0.12, 16, 16); const blueberryMat = new THREE.MeshPhongMaterial({ color: 0x1a237e, shininess: 10 }); return new THREE.Mesh(blueberryGeo, blueberryMat); }
